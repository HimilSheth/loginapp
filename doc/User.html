def self.from_omniauth(auth)
  #byebug
  create! do |user|
    user.email = auth["info"]["email"]
    user.name = auth["info"]["first_name"]
    user.password = auth.credentials.token
    user.password_confirmation = auth.credentials.token
  end
end
